namespace = ACS

# Vee spawns in a random ruler around the world. No players this time
ACS.0001 = {
	type = empty
	hidden = yes

	immediate = {
		set_global_variable = {
			name = ancestors_startup
			value = yes
		}
		random_ruler = {
			limit = { is_ai = yes }
			trigger_event = ACS.0002
		}
	}
}

ACS.0002 = {
	type = empty
	hidden = yes

	immediate = {
		root = {	
			create_character = {
				location = root.capital_province
				name = "Vee"
				age = 0
				culture = root.culture
				faith = root.faith
				dynasty = none
				gender = female
				save_scope_as = vee

				trait = brave
				trait = sharingan
				trait = lustful
				trait = ambitious
				trait = fecund
				trait = beauty_good_3
				trait = intellect_good_3
				trait = physique_good_3
				trait = mystic_3
				trait = athletic
				trait = reincarnation
				trait = wild_oat
				trait = syrena
			}
			add_courtier = scope:vee
			scope:vee = {
				add_character_modifier = {
					modifier = sharingan_promise
				}
			}
		}
	}
}

# On birth, mark a character until adulthood
ACS.0003 = {
	type = empty
	hidden = yes

	immediate = {
		# Evaluates sharingan
		if = {
			limit = {			
				OR = {
					mother = {
						has_character_modifier = sharingan_promise
					}
					real_father = {
						has_character_modifier = sharingan_promise
					}
				}
			}
			add_character_modifier = {
				modifier = sharingan_promise
			}
		}
		# Evaluates byakugan
		if = {
			limit = {			
				OR = {
					mother = {
						has_character_modifier = byakugan_promise
					}
					real_father = {
						has_character_modifier = byakugan_promise
					}
				}
			}
			add_character_modifier = {
				modifier = byakugan_promise
			}
		}
	}
}

# Decide byakugan or sharingan inheritance
ACS.0004 = {
	type = empty
	hidden = yes

	immediate = {
			# Incompatible dojutsu types
			if = {
				limit = {
					OR = {
						AND = {
							mother = {
								has_character_modifier = sharingan_promise
							}
							real_father = {
								has_character_modifier = byakugan_promise
							}
						}
						AND = {
							real_father = {
								has_character_modifier = sharingan_promise
							}
							mother = {
								has_character_modifier = byakugan_promise
							}
						}
					}				
				}
				add_trait = possessed_1
				add_trait = hunchbacked
				add_trait = infertile
				add_trait = bleeder
				add_trait = physique_bad_1
				add_trait = disfigured
				add_trait = intellect_good_3
				add_trait = blind
				add_trait = impotent
			}
			# Checking mother sharingan
			if = {
				limit = {
					mother = {
						has_character_modifier = sharingan_promise
					}
				}
				add_character_modifier = {
					modifier = sharingan_promise
				}
				random_list = {
					3 = {
						add_trait = sharingan
					}
					97 = {}
				}
			}
			# Checking real_father sharingan
			if = {
				limit = {
					real_father = {
						has_character_modifier = sharingan_promise
					}
				}
				add_character_modifier = {
					modifier = sharingan_promise
				}
				random_list = {
					3 = {
						add_trait = sharingan
					}
					97 = {}
				}
			}
			# Checking real_father byakugan
			if = {
				limit = {
					real_father = {
						has_character_modifier = byakugan_promise
					}
				}
				add_character_modifier = {
					modifier = byakugan_promise
				}
				random_list = {
					3 = {
						add_trait = byakugan
					}
					97 = {}
				}
			}
			# Checking mother byakugan
			if = {
				limit = {
					mother = {
						has_character_modifier = byakugan_promise
					}
				}
				add_character_modifier = {
					modifier = byakugan_promise
				}
				random_list = {
					3 = {
						add_trait = byakugan
					}
					97 = {}
				}
			}
	}
}

# Khalga is coming
ACS.0005 = {
	type = empty
	hidden = yes

	immediate = {
		debug_log = "SELECTING CACHUDO"
		random_ruler = {
			limit = {
				is_married = yes
				any_spouse = {
					age < 40
				}
				is_female = no
			}
			trigger_event = ACS.0006
		}
	}
}

ACS.0006 = {
	type = empty
	hidden = yes

	immediate = {
		debug_log = "selecting waifu"
		random_spouse = {
			limit = {
				age < 40
			}
			save_scope_as = khalga_target
		}
		create_character = {
			location = scope:khalga_target.location
			name = "Khalga"
			age = 26
			culture = scope:khalga_target.culture
			faith = scope:khalga_target.faith
			dynasty = none
			gender = male
			save_scope_as = khalga

			trait = brave
			trait = byakugan
			trait = lustful
			trait = ambitious
			trait = fecund
			trait = beauty_good_3
			trait = intellect_good_3
			trait = physique_good_3
			trait = mystic_3
			trait = athletic
			trait = reincarnation
			trait = wild_oat
			trait = zeus
		}
		scope:khalga_target = {
			make_pregnant = {
				father = scope:khalga
			}
			set_relation_lover = scope:khalga
		}
		# Khalga dies
		scope:khalga = {
			add_character_modifier = {
				modifier = byakugan_promise
			}
			trigger_event = {
				id = ACS.0007
				months = 11
			}
		}
	}
}
