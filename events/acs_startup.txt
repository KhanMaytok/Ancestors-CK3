namespace = ACS

# Select a random ruler. only AI.
ACS.0001 = {
	type = empty
	hidden = yes

	trigger = {
		NOT = { has_global_variable = ancestors_startup }
	}

	immediate = {
		set_global_variable = {
			name = ancestors_startup
			value = yes
		}
		random_ruler = {
			limit = { is_ai = yes }
			trigger_event = ACS.0002
		}
	}
}

# Spawn Vee at selected ruler
ACS.0002 = {
	type = empty
	hidden = yes

	immediate = {
		debug_log = "First Vee"
		root = {	
			create_character = {
				location = root.capital_province
				name = "Vee"
				age = 0	
				culture = root.culture
				faith = root.faith
				dynasty = none
				gender = female
				save_scope_as = vee

				trait = brave
				trait = sharingan
				trait = lustful
				trait = ambitious
				trait = fecund
				trait = beauty_good_3
				trait = intellect_good_3
				trait = physique_good_3
				trait = mystic_3
				trait = athletic
				trait = reincarnation
				trait = wild_oat
				trait = syrena
				trait = vee_blood
			}
			add_courtier = scope:vee
		}
	}
}

# Sharingan or byakugan on adulthood
ACS.0003 = {
	type = empty
	hidden = yes

	immediate = {
		# Evaluates sharingan
		if = {
			limit = {			
				has_trait = vee_blood
			}
			random_list = {
				93 = {}
				7 = {
					add_trait = sharingan
				}
			}
		}
		# Evaluates byakugan
		if = {
			limit = {			
				has_trait = zeus_blood
			}
			random_list = {
				93 = {}
				7 = {
					add_trait = byakugan
				}
			}
		}
		# Incompatible two traits
		if = {
			limit = {			
				has_trait = sharingan
				has_trait = byakugan
			}
			add_trait = possessed_1
			add_trait = hunchbacked
			add_trait = infertile
			add_trait = bleeder
			add_trait = physique_bad_1
			add_trait = disfigured
			add_trait = intellect_good_3
			add_trait = blind
			add_trait = impotent
		}
	}
}

# Khalga is coming. Select suitable ruler
ACS.0004 = {
	type = empty
	hidden = yes

	immediate = {
		debug_log = "SELECTING CACHUDO"
		random_ruler = {
			limit = {
				is_married = yes
				any_spouse = {
					age < 40
				}
				is_female = no
			}
			trigger_event = ACS.0005
		}
	}
}

# Select ruler spouse
ACS.0005 = {
	type = empty
	hidden = yes

	immediate = {
		debug_log = "selecting waifu"
		random_spouse = {
			limit = {
				age < 40
			}
			save_scope_as = khalga_target
		}
		create_character = {
			location = scope:khalga_target.location
			name = "Khalga"
			age = 26
			culture = scope:khalga_target.culture
			faith = scope:khalga_target.faith
			dynasty = none
			gender = male
			save_scope_as = khalga

			trait = brave
			trait = byakugan
			trait = lustful
			trait = ambitious
			trait = fecund
			trait = beauty_good_3
			trait = intellect_good_3
			trait = physique_good_3
			trait = mystic_3
			trait = athletic
			trait = reincarnation
			trait = wild_oat
			trait = zeus
			trait = zeus_blood
		}
		scope:khalga_target = {
			make_pregnant = {
				father = scope:khalga
			}
			set_relation_lover = scope:khalga
		}
		# Khalga dies
		scope:khalga = {
			trigger_event = {
				id = ACS.0006
				days = { 180 200 }
			}
		}
	}
}

# Khalga dies
ACS.0006 = {
	type = empty
	hidden = yes

	immediate = {
		debug_log = "killing khalga"
		death = {
			death_reason = death_hunting_accident
		}
	}
}
